# ==============================================================================
# Reconfiguring the Ideal Order: Code Repository
# ==============================================================================
#
# This repository contains the computational code for:
#
# Hamilton, G. & Sorensen, E.P. (2026). "Reconfiguring the ideal order:
# ablation and field formation in the twentieth-century Nigerian novel in
# English." Digital Humanities Quarterly, 20(1).
# https://dhq.digitalhumanities.org/vol/20/1/000858/000858.html
#
# Overview
# --------
# This code implements a computational methodology for measuring each text's
# structural contribution to the semantic field of a literary corpus. The
# "disruption score" framework treats a corpus as an integrated semantic space
# constructed through GloVe word embeddings, then employs ablation testing to
# assess how that space changes when individual texts are removed. Procrustes
# analysis quantifies the geometric difference between full and ablated
# embeddings, and an iterative protocol with consistency-based exponential
# decay filtering distinguishes genuine semantic disruption from algorithmic
# noise.
#
# Sections
# --------
#  1.  Setup and Configuration
#  2.  Load Required Libraries
#  3.  Build Corpus
#  4.  Tokenization and Per-Text Word Embeddings
#  5.  Full Corpus Word Embedding
#  6.  Cosine Similarity Adjacency Matrix
#  7.  Recast Cosine Similarity Networks (n-1)
#  8.  Procrustes Distance Between Full and Recast Networks
#  9.  Ablation Testing and Procrustes Analysis (n-1 Embeddings)
#  10. Cosine Similarity Network
#  11. Network Edge Analysis
#  12. Sum Weight of Out-Edges
#  13. Final Analysis — Disruption Scores, Tables 1 & 2, and Network
#  14. Network Analysis — Eigenvector Centrality and Influence
#  15. Weight Optimization
#  16. Diagnostic Plots — Metric Relationships
#  17. Validation — Text Length vs. Disruption Score
#
# Key Functions:
#   find_shared_words()                 - Identifies anchor vocabulary shared
#                                         across all texts in the corpus
#   check_token_sequence_sequentially() - Locates each text's token sequence
#                                         within the full corpus token stream
#   remove_token_sequence()             - Removes a text's tokens to produce
#                                         the ablated corpus
#   calculate_recast_similarities()     - Computes n-1 cosine similarity
#                                         networks for each text
#   find_best_weights()                 - Optimises composite scoring weights
#                                         across disruption, eigenvector
#                                         centrality, and network intersection
#

# SECTION 4: Tokenization and Per-Text Word Embeddings
# ==============================================================================
#
# For each text in the corpus subset, this section:
# 1. Tokenizes and cleans the text
# 2. Constructs a feature co-occurrence matrix (FCM)
# 3. Trains a GloVe word embedding model
# Word vectors and token lists are stored as named objects in the environment.

start_time <- Sys.time()
counter    <- 0

for (i in 1:ndoc(corpus_sub_sub)) {
    counter <- counter + 1
    print(paste0("Tokenizing... ", counter, " of ", ndoc_ss))

    # Define anchor words to preserve during stopword removal
    base_anchor <- c("the", "and", "of", "in", "to", "is", "was",
                     "said", "had", "he", "she", "they")

    # Replace hyphens and separate compound words
    tnh <- gsub("-", " ", corpus_sub_sub[i])

    # Tokenize and clean
    a.t_c <- quanteda::tokens(tnh,
                              remove_punct   = TRUE,
                              remove_symbols = TRUE,
                              remove_numbers = TRUE)
    a.t_c_lo <- tokens_tolower(a.t_c)

    # Build modified stopword list (preserving anchor words)
    english_stops  <- stopwords("english")
    english_stops  <- english_stops[!english_stops %in% base_anchor]
    stoplist_legacy <- c(letters, LETTERS)

    # Remove stopwords
    gat_rm <- tokens_remove(a.t_c_lo, c(english_stops, stoplist_legacy))

    # Convert to format needed for FCM
    gat_list   <- as.list(gat_rm)
    gat_mo     <- paste(unlist(gat_list), collapse = " ")
    gat_mo_tox <- quanteda::tokens(gat_mo)

    # Store tokenized text
    a.nam <- paste0(corpus_sub_sub$short_name[i], "_tokslist")
    assign(a.nam, gat_mo_tox)

    # Create FCM with distance-weighted context window
    print(paste0("Creating word vectors..."))
    b.fcmat <- fcm(gat_mo_tox, context = "window", count = "weighted",
                   weights = 1 / (1:5), tri = TRUE)

    # Train GloVe word embedding model
    set.seed(42)
    glove          <- GlobalVectors$new(rank = 50, x_max = 10)
    b.wv_main      <- glove$fit_transform(b.fcmat, n_iter = 15,
                                          convergence_tol = 0.01, n_threads = 8)
    b.wv_context   <- glove$components
    b.word_vectors <- b.wv_main + t(b.wv_context)

    # Store word vectors
    b.nam <- paste0(corpus_sub_sub$short_name[i], "_vex")
    assign(b.nam, b.word_vectors)

    print(paste0("Completed text ", i))
}

# ==============================================================================
# SECTION 7: Recast Cosine Similarity Networks (n-1)
# ==============================================================================
#
# For each text, calculates an adjacency matrix for the corpus minus that
# text, saving results to CSV files and to the global environment.
# Note: This procedure is computationally intensive. The function prompts for
# a range of texts to process in a single run to manage heat and memory.

calculate_recast_similarities <- function(dfm_aka, start_pos, end_pos) {

    total_texts <- ndoc(dfm_aka)
    if (is.na(start_pos) || is.na(end_pos) ||
        start_pos < 1 || end_pos > total_texts ||
        start_pos > end_pos) {
        stop(paste("Please enter valid positions between 1 and", total_texts))
    }

    library(quanteda)
    library(quanteda.textstats)
    library(igraph)
    library(Matrix)

    print(paste("Processing texts", start_pos, "to", end_pos,
                "of", total_texts, "total texts"))

    start_time <- Sys.time()

    for (i in start_pos:end_pos) {
        current_shortname <- docvars(dfm_aka[i, ], "short_name")
        print(paste("Processing recast", i, "of", total_texts, "-", current_shortname))

        reduced_dfm <- dfm_aka[-i, ]

        reduced_adj_matrix <- matrix(0,
                                     nrow = ndoc(reduced_dfm),
                                     ncol = ndoc(reduced_dfm))
        rownames(reduced_adj_matrix) <- colnames(reduced_adj_matrix) <- docnames(reduced_dfm)

        doc_authors  <- docvars(reduced_dfm, "author")
        doc_years    <- docvars(reduced_dfm, "year")
        current_year <- docvars(dfm_aka[i, ], "year")

        for (j in 1:ndoc(reduced_dfm)) {
            row_results <- numeric(ndoc(reduced_dfm))

            for (k in 1:ndoc(reduced_dfm)) {
                if (j != k &&
                    doc_authors[j] != doc_authors[k] &&
                    doc_years[k] > current_year) {

                    sim_score <- as.numeric(textstat_simil(reduced_dfm[j, ],
                                                           reduced_dfm[k, ],
                                                           method = "cosine",
                                                           margin = "documents"))
                    row_results[k] <- sim_score
                }
            }
            reduced_adj_matrix[j, ] <- row_results
        }

        filename <- paste0("recast_matrix_", current_shortname, ".csv")
        write.csv(reduced_adj_matrix, filename)
        print(paste("Saved matrix for", current_shortname, "to", filename))

        assign(paste0(current_shortname, "_rcm"), reduced_adj_matrix, envir = .GlobalEnv)

        if (i < end_pos) {
            print("Pausing before processing next batch...")
            Sys.sleep(30)
        }
    }

    end_time  <- Sys.time()
    exec_time <- end_time - start_time
    print(paste("Analysis complete. Total execution time:",
                round(exec_time, 2), "minutes"))
}

# Run the function: set start_pos and end_pos to the range of texts to process.
# Processing the full corpus in one run is intensive; running in batches of ~5
# is recommended to manage memory and heat.
# Example: calculate_recast_similarities(dfm_aka, start_pos = 1, end_pos = 5)
calculate_recast_similarities(dfm_aka, start_pos = 1, end_pos = ndoc(dfm_aka))

# ==============================================================================
# SECTION 8: Procrustes Distance Between Full and Recast Networks
# ==============================================================================
#
# Uses the protest() function from the vegan package to calculate the
# difference between each recast adjacency matrix and the corresponding
# subset of the full adjacency matrix.

library(vegan)

n_rcms      <- length(pull_rcms)
differences <- rep(NA_real_, n_rcms)
p_values    <- rep(NA_real_, n_rcms)

for (i in 1:n_rcms) {
    rcm      <- pull_rcms[[i]]
    full_sub <- adj_matrix[-i, -i]

    # Skip if either matrix has too few non-zero values for a meaningful comparison
    if (sum(rcm != 0) < 2 || sum(full_sub != 0) < 2) {
        warning(paste("Skipping text", i,
                      "— insufficient non-zero values for Procrustes comparison"))
        next
    }

    tryCatch({
        prc            <- protest(rcm, full_sub, permutations = 999)
        differences[i] <- prc$ss
        p_values[i]    <- prc$signif
    }, error = function(e) {
        warning(paste("Error processing text", i, ":", e$message))
    })
}

text_names <- rownames(adj_matrix)[1:n_rcms]
ress <- data.frame(
    text       = text_names,
    difference = differences,
    p_value    = p_values
)

# Sort by difference to identify most influential texts (NAs placed last)
ress <- ress[order(-ress$difference, na.last = TRUE), ]
ress


# ==============================================================================
# SECTION 9: Ablation Testing and Procrustes Analysis (n-1 Embeddings)
# ==============================================================================
#
# Identifies anchor words shared across all texts, then for each text:
# 1. Removes that text's tokens from the full corpus token sequence
# 2. Retrains a GloVe embedding on the reduced corpus (5 iterations)
# 3. Aligns the full and reduced embeddings using Procrustes analysis
# 4. Applies an exponential decay function to penalise high stochastic variance
# Results are stored in result_dfr.

library(quanteda)
library(text2vec)
library(vegan)

# -----------------------------------------------------------------------------
# Find words shared across every text in the corpus
# -----------------------------------------------------------------------------

find_shared_words <- function(pull_tokslist) {
    shared_words <- attr(pull_tokslist[[1]], "types")
    for (i in 2:length(pull_tokslist)) {
        current_words <- attr(pull_tokslist[[i]], "types")
        shared_words  <- intersect(shared_words, current_words)
    }
    return(shared_words)
}

anchor_words <- find_shared_words(pull_tokslist)
print(paste0("Number of anchor words to align word embeddings: ", length(anchor_words)))

# Preprocess a text sequence
preprocess_sequence <- function(sequence, anchor_words) {
    sequence_no_hyphen <- gsub("-", " ", sequence)
    tokens_seq    <- quanteda::tokens(sequence_no_hyphen,
                                      remove_punct   = TRUE,
                                      remove_symbols = TRUE,
                                      remove_numbers = TRUE)
    tokens_seq_lo <- tokens_tolower(tokens_seq)
    return(tokens_seq_lo)
}

# Locate the start and end indices of each text's token sequence within full_toks
check_token_sequence_sequentially <- function(full_toks, text_toks_list,
                                              seq_length        = 15,
                                              preprocess_length = 25) {
    tokens_vec    <- as.list(full_toks)[[1]]
    start_indices <- c()
    end_indices   <- c()

    for (text_toks in text_toks_list) {
        actual_length <- length(text_toks[[1]])
        print(paste("Processing sequence of length:", actual_length))

        start_seq <- as.list(text_toks)[[1]][1:min(preprocess_length, actual_length)]
        end_seq   <- as.list(text_toks)[[1]][max(1, actual_length - preprocess_length + 1):actual_length]

        start_seq_text <- paste(start_seq, collapse = " ")
        end_seq_text   <- paste(end_seq,   collapse = " ")

        start_seq_processed <- preprocess_sequence(start_seq_text, anchor_words)
        end_seq_processed   <- preprocess_sequence(end_seq_text,   anchor_words)

        start_seq_final <- as.list(start_seq_processed)[[1]][
            1:min(seq_length, length(start_seq_processed[[1]]))]
        end_seq_final <- as.list(end_seq_processed)[[1]][
            max(1, length(end_seq_processed[[1]]) - seq_length + 1):length(end_seq_processed[[1]])]

        print("Looking for sequence...")
        print(paste0("Start: ", paste(start_seq_final, collapse = " ")))
        print(paste0("End: ",   paste(end_seq_final,   collapse = " ")))

        start_matches <- which(sapply(1:(length(tokens_vec) - length(start_seq_final) + 1), function(i) {
            all(tokens_vec[i:(i + length(start_seq_final) - 1)] == start_seq_final)
        }))

        if (length(start_matches) == 0) stop("Start sequence not found in full_toks")
        start_index <- start_matches[1]

        search_range_start <- start_index
        search_range_end   <- min(start_index + actual_length + 100,
                                  length(tokens_vec) - length(end_seq_final) + 1)

        end_matches <- which(sapply(search_range_start:search_range_end, function(i) {
            all(tokens_vec[i:(i + length(end_seq_final) - 1)] == end_seq_final)
        }))

        if (length(end_matches) == 0) stop("End sequence not found in full_toks")
        end_index <- search_range_start + end_matches[length(end_matches)] - 1 +
                     length(end_seq_final) - 1

        range_length <- end_index - start_index + 1
        if (range_length > actual_length * 1.2) {
            warning(paste("Suspicious range length:", range_length,
                          "for sequence of length:", actual_length))
            next
        }

        start_indices <- c(start_indices, start_index)
        end_indices   <- c(end_indices,   end_index)
    }

    print(paste("Found sequence range:",
                paste(start_indices, collapse = ","), "to",
                paste(end_indices,   collapse = ",")))

    return(list(start_indices = start_indices, end_indices = end_indices))
}

# Remove a token sequence by index positions
remove_token_sequence <- function(tokens_vec, start_indices, end_indices) {
    remove_indices <- unlist(mapply(function(start, end) start:end,
                                   start_indices, end_indices))
    tokens_vec <- tokens_vec[-remove_indices]
    return(tokens_vec)
}

        for (z in 1:5) {
            print(paste0("Iteration ", z, "/5"))
            m.glove        <- GlobalVectors$new(rank = 50, x_max = 10)
            m.a.wv_main    <- m.glove$fit_transform(m.a.fcmat, n_iter = 3,
                                                    convergence_tol = 0.01,
                                                    n_threads = 8)
            m.a.wv_context <- m.glove$components
            m.a.word_vectors <- m.a.wv_main + t(m.a.wv_context)

            ma_vectors <- scale(m.a.word_vectors[anchor_words, ])
            a_vectors  <- scale(a.word_vectors[anchor_words, ])

            alignment <- procrustes(ma_vectors, a_vectors)
            ssq       <- c(ssq, alignment$ss)
        }

                       comp_score, comp_w_decay)
        )

    })

# SECTION 15: Weight Optimization
# ==============================================================================
#
# Searches a grid of weights to find the combination that best balances the
# contribution of disruption, eigenvector centrality, and high-disruption
# intersections to the composite score.

library(scales)
library(dplyr)

calculate_total <- function(weights, df) {
    rowSums(sweep(df[, c("disruption", "eigenvector", "hd_intersects")],
                  2, weights, FUN = "*"))
}

weight_grid <- expand.grid(
    disruption    = seq(0.1, 0.9, by = 0.1),
    eigenvector   = seq(0.1, 0.9, by = 0.1),
    hd_intersects = seq(0.1, 0.9, by = 0.1)
)

find_best_weights <- function(weight_grid, df) {
    best_weights    <- NULL
    best_similarity <- -Inf

    for (i in 1:nrow(weight_grid)) {
        weights    <- as.numeric(weight_grid[i, ])
        df$total   <- calculate_total(weights, df)
        similarity <- sum(
            cor(df$disruption,    df$total, use = "complete.obs"),
            cor(df$eigenvector,   df$total, use = "complete.obs"),
            cor(df$hd_intersects, df$total, use = "complete.obs")
        )
        if (!is.na(similarity) && similarity > best_similarity) {
            best_similarity <- similarity
            best_weights    <- weights
        }
    }
    return(best_weights)
}

best_weights <- find_best_weights(weight_grid, centrality_measures_norm)
